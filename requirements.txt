# ============================================================================
# RASPBERRY PI INSTALLATION NOTES
# ============================================================================
#
# BEFORE INSTALLING PYTHON PACKAGES, RUN THESE SYSTEM COMMANDS:
#
# 1. Update system packages:
#    sudo apt update && sudo apt upgrade -y
#
# 2. Install system dependencies:
#    sudo apt install -y python3-pip python3-dev python3-venv
#    sudo apt install -y portaudio19-dev libasound2-dev
#    sudo apt install -y libatlas-base-dev libopenblas-dev
#    sudo apt install -y cmake build-essential
#    sudo apt install -y libhdf5-dev libhdf5-serial-dev
#    sudo apt install -y libqtgui4 libqt4-test
#    sudo apt install -y libavcodec-dev libavformat-dev libswscale-dev
#    sudo apt install -y libgtk2.0-dev pkg-config
#    sudo apt install -y libjpeg-dev libtiff5-dev libpng-dev
#    sudo apt install -y libv4l-dev v4l-utils
#
# 3. Install Python packages:
#    pip install -r requirements.txt
#
# 4. For face recognition optimization on Raspberry Pi:
#    pip install --no-cache-dir dlib
#    pip install --no-cache-dir face-recognition
#
# 5. Test camera access:
#    sudo usermod -a -G video $USER
#    (logout and login again)
#
# ============================================================================
# OPTIONAL OPTIMIZATIONS FOR RASPBERRY PI
# ============================================================================
#
# For better performance, you can install these optimized versions:
#
# 1. Lightweight OpenCV (smaller footprint):
#    pip uninstall opencv-python
#    pip install opencv-python-headless
#
# 2. Pre-compiled wheels for ARM (faster installation):
#    pip install --extra-index-url https://www.piwheels.org/simple/ numpy
#    pip install --extra-index-url https://www.piwheels.org/simple/ scipy
#
# 3. Reduce memory usage during installation:
#    export MAKEFLAGS="-j1"
#    pip install --no-cache-dir -r requirements.txt
#
# ============================================================================
# TROUBLESHOOTING COMMON RASPBERRY PI ISSUES
# ============================================================================
#
# 1. If PyAudio installation fails:
#    sudo apt install portaudio19-dev python3-pyaudio
#    pip install pyaudio
#
# 2. If dlib compilation takes too long or fails:
#    sudo apt install libdlib-dev
#    pip install --no-cache-dir dlib
#
# 3. If OpenCV installation fails:
#    sudo apt install python3-opencv
#    pip install opencv-contrib-python-headless
#
# 4. If torch installation is too slow:
#    pip install torch torchvision --extra-index-url https://download.pytorch.org/whl/cpu
#
# 5. Memory issues during installation:
#    sudo dphys-swapfile swapoff
#    sudo nano /etc/dphys-swapfile  # Set CONF_SWAPSIZE=1024
#    sudo dphys-swapfile setup
#    sudo dphys-swapfile swapon
